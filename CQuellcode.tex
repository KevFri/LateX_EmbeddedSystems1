    uint16_t ui16_i=0;
    while(u16milliseconds){
        for (ui16_i=0;ui16_i<331;ui16_i++){    //1 ms delay loop
            __asm__ volatile("nop \n\t"
                             "nop \n\t"

        }//for
        u16milliseconds--;
    }//while
}
